修复 "Unexpected token '<', "

已连接到Salesforce

问题分析

当网页尝试向Salesforce提交数据时，出现错误："提交失败: Unexpected token '<', "

该错误表明Salesforce返回了HTML内容而非预期的JSON响应，通常由以下原因导致：

API端点配置错误

使用了相对路径而不是绝对URL，导致请求被重定向到登录页面

缺少身份验证

未提供有效的Session ID或未配置访客用户权限

CORS配置问题

Salesforce未配置允许跨域请求，导致浏览器阻止响应

解决方案演示

以下表单已修复API端点和身份验证问题

配置说明

1. 在Salesforce中设置站点并启用Web-to-Lead

2. 为访客用户分配以下权限：

  • Lead对象的创建权限
  • 相关字段的读写权限
  • LeadService Apex类的执行权限

Apex类修复 (LeadService.cls)
global without sharing class LeadService {
    
    // 添加CORS支持
    @HttpPost
    global static Response createLead() {
        // 设置响应头
        RestResponse res = RestContext.response;
        if(res == null) {
            res = new RestResponse();
            RestContext.response = res;
        }
        res.addHeader('Access-Control-Allow-Origin', '*');
        res.addHeader('Access-Control-Allow-Methods', 'POST');
        res.addHeader('Content-Type', 'application/json');
        
        Response responseObj = new Response();
        
        try {
            // 从请求中获取数据
            RestRequest req = RestContext.request;
            Map<String, Object> requestBody = 
                (Map<String, Object>)JSON.deserializeUntyped(req.requestBody.toString());
            
            // 获取字段值...
            
            // 检查重复记录
            if (hasDuplicates(phone, email)) {
                responseObj.success = false;
                responseObj.isDuplicate = true;
                responseObj.message = '系统中已存在相同联系方式的记录';
                return responseObj;
            }
            
            // 创建新的Lead记录...
            
            responseObj.success = true;
            responseObj.message = '记录创建成功';
            responseObj.leadId = newLead.Id;
            
        } catch (Exception e) {
            responseObj.success = false;
            responseObj.message = '错误: ' + e.getMessage();
        }
        
        return responseObj;
    }
    
    // 检查重复记录方法保持不变...
    
    // 响应类
    global class Response {
        global Boolean success;
        global Boolean isDuplicate;
        global String message;
        global String leadId;
        
        global Response() {
            success = false;
            isDuplicate = false;
            message = '';
            leadId = '';
        }
    }
}
JavaScript修复 (HTML页面)
// 动态获取Salesforce实例域名
const getSalesforceBaseUrl = () => {
    if(window.location.host.includes('visual.force.com')) {
        return window.location.origin.replace('.visual.force.com', 
            '.my.salesforce.com');
    }
    return window.location.origin;
};

// 使用动态域名构建API端点
const SALESFORCE_API_ENDPOINT = 
    getSalesforceBaseUrl() + '/services/apexrest/LeadService';

// 表单提交处理
submitBtn.addEventListener('click', async function() {
    // ...验证逻辑...
    
    try {
        // 调用Salesforce REST API
        const response = await fetch(SALESFORCE_API_ENDPOINT, {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify(formData)
        });
        
        // 检查响应内容类型
        const contentType = response.headers.get('content-type');
        let result;
        
        if (contentType && contentType.includes('application/json')) {
            result = await response.json();
        } else {
            // 处理非JSON响应
            const text = await response.text();
            throw new Error(`无效响应: ${text.substring(0, 100)}`);
        }
        
        if (response.ok && result.success) {
            // 提交成功...
        } else {
            // 处理错误...
        }
    } catch (error) {
        // 显示错误消息
        formStatus.textContent = '提交失败: ' + error.message;
        formStatus.className = 'status-message status-error';
        submitText.textContent = '提交咨询';
        submitLoader.style.display = 'none';
    }
});
